alerts = driver.find_elements(By.CLASS_NAME, "alert-class")  # "alert-class" ko aapke HTML ke hisaab se change karna hoga

for alert in alerts:
    # Har ek alert ke andar doosra link retrieve karna
    alert_links = alert.find_elements(By.CLASS_NAME, "fxc-gcflink-link")  # Yeh selector aapke HTML ke hisaab se change karna hoga
    if len(alert_links) >= 2:  # Agar do links hain
        second_link = alert_links[1]  # Second link ko retrieve karna
        second_link.click()  # Second link par click karna
        # Wait for next dashboard to load
        time.sleep(8)
        # Capture screenshot of the next dashboard
        driver.save_screenshot(f'next_dashboard_{index}.png')
        # Go back to the previous page (the alert details)
        driver.back()
        # Wait for page to load
        time.sleep(5)

# Quit the driver
driver.quit()
