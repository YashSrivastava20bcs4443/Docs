# Find all rows containing alerts
rows = driver.find_elements(By.XPATH, "//div[@class='fxc-gc-row']")

# Iterate through each row containing an alert
for index, row in enumerate(rows, 1):
    try:
        # Find the link within the third column of the row
        third_column = row.find_element(By.XPATH, ".//div[@class='fxc-gc-cell fxc-gc-columncell_0_2']//a[@class='fxc-gcflink-link']")
        
        # Click on the link
        third_column.click()
        
        # Wait for the page to load
        time.sleep(5)  # Adjust the wait time as needed
        
        # Take a screenshot of the page
        driver.save_screenshot(f"screenshot_{index}.png")
        
        # Go back to the previous page (the alert details)
        driver.back()
        
        # Wait for the page to load
        time.sleep(5)  # Adjust the wait time as needed
    except Exception as e:
        print(f"An error occurred: {e}")

# Quit the driver
driver.quit()
