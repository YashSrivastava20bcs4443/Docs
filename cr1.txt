import os

top_alert_row = driver.find_element(By.XPATH, "(//div[contains(@class, 'fxc-gc-row')])[1]")

try:
    # Find the third column in the top alert row
    third_column = top_alert_row.find_element(By.XPATH, ".//div[contains(@class, 'fxc-gc-cell')][3]//a[@class='fxc-gcflink-link']")
    
    # Scroll to the element
    actions = ActionChains(driver)
    actions.move_to_element(third_column).perform()
    
    # Wait for the element to be clickable
    WebDriverWait(driver, 10).until(EC.element_to_be_clickable((By.XPATH, ".//div[contains(@class, 'fxc-gc-cell')][3]//a[@class='fxc-gcflink-link']")))
    
    # Get the text of the link
    link_text = third_column.text
    
    # Modify the link text to remove invalid characters for a filename
    filename = link_text.replace('/', '_').replace(':', '_').replace('?', '_').replace('&', '_')
    
    # Execute JavaScript to click on the third column
    driver.execute_script("arguments[0].click();", third_column)
    
    # Wait for the new page to load
    time.sleep(5)  # You might need to adjust the wait time
    
    # Take a screenshot of the page
    driver.save_screenshot(f"{filename}_screenshot.png")
    
    # Go back to the previous page (the alert details)
    driver.back()
    
    # Wait for the page to load
    time.sleep(5)  # You might need to adjust the wait time

    # Find column 1 in the top alert row
    column_1 = top_alert_row.find_element(By.XPATH, ".//div[contains(@class, 'fxc-gc-cell')][1]//a[@class='fxc-gcflink-link']")
    
    # Get the text of column 1
    column_1_text = column_1.text
    
    # Find column 2 in the top alert row
    column_2 = top_alert_row.find_element(By.XPATH, ".//div[contains(@class, 'fxc-gc-cell')][2]/span[@class='ext-severity-text']")
    
    # Get the text of column 2
    column_2_text = column_2.text
    
    # Write the data to a text file
    with open("alert_data.txt", "w") as file:
        file.write(f"Column 1: {column_1_text}\n")
        file.write(f"Column 2: {column_2_text}\n")
        
except Exception as e:
    print(f"An error occurred: {str(e)}")
