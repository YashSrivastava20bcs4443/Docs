from selenium.webdriver.common.by import By
import time

# Find all alert elements
alert_elements = driver.find_elements(By.XPATH, "//div[@data-automation-id='detailsList']/div[contains(@role, 'listitem')]")

# Iterate through each alert element
for index, alert_element in enumerate(alert_elements, 1):
    # Find the second link in the current alert element
    second_link = alert_element.find_element(By.XPATH, ".//a[@class='fxc-gcflink-link'][2]")
    
    # Click on the second link
    second_link.click()
    
    # Wait for the next dashboard to load
    time.sleep(8)
    
    # Capture screenshot of the current dashboard
    driver.save_screenshot(f'next_dashboard_{index}.png')

    # Go back to the previous page (the alert details)
    driver.back()

    # Wait for page to load
    time.sleep(5)

# Quit the driver
driver.quit()
