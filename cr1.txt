# Find all rows containing alerts
rows = driver.find_elements(By.XPATH, "//div[@class='fxc-gc-row']")

# Iterate through each row
for index, row in enumerate(rows, 1):
    print(f"Processing Alert {index}")

    try:
        # Find the third column in the row
        third_column = row.find_element(By.XPATH, ".//div[@class='fxc-gc-cell fxc-gc-columncell_0_2']")

        # Scroll to the element
        driver.execute_script("arguments[0].scrollIntoView();", third_column)

        # Wait for the element to be clickable
        wait = WebDriverWait(driver, 10)
        third_column = wait.until(EC.element_to_be_clickable((By.XPATH, ".//div[@class='fxc-gc-cell fxc-gc-columncell_0_2']")))

        # Click on the third column
        third_column.click()

        # Wait for the page to load
        time.sleep(5)  # Adjust the wait time as needed

        # Take a screenshot
        driver.save_screenshot(f"alert_{index}_screenshot.png")

        # Go back to the previous page
        driver.back()

        # Wait for the page to load
        time.sleep(5)  # Adjust the wait time as needed

    except Exception as e:
        print(f"Error occurred: {e}")

# Close the WebDriver
driver.quit()
