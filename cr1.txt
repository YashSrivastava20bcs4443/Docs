from selenium.webdriver.common.by import By
import time

# Find all alert elements
alert_elements = driver.find_elements(By.XPATH, "//div[@data-automation-id='detailsList']/div[contains(@role, 'listitem')]")

# Iterate through each alert
for index, alert_element in enumerate(alert_elements, 1):
    # Click on the alert element to view details
    alert_element.click()

    # Wait for alert details to load
    time.sleep(5)

    # Find the link to navigate to the next dashboard within the current alert element
    link_to_next_dashboard = alert_element.find_element(By.XPATH, ".//a[contains(@class, 'next-dashboard-link')]")
    
    # Get the URL of the next dashboard
    next_dashboard_url = link_to_next_dashboard.get_attribute('href')
    
    # Open the next dashboard
    driver.get(next_dashboard_url)
    
    # Wait for the next dashboard to load
    time.sleep(5)
    
    # Capture screenshot of the next dashboard
    driver.save_screenshot(f'next_dashboard_{index}.png')

    # Go back to the previous page (alert details)
    driver.back()

    # Wait for page to load
    time.sleep(5)

# Quit the driver
driver.quit()
