alert_links = driver.find_elements(By.CLASS_NAME, "fxc-gcflink-link")

# Iterate through each alert link
for index, alert_link in enumerate(alert_links, 1):
    try:
        # Construct the JavaScript to click on the anchor tag
        js_script = "arguments[0].click();"
        
        # Execute the JavaScript to click on the anchor tag
        driver.execute_script(js_script, alert_link)
        
        # Wait for the link to be clickable
        WebDriverWait(driver, 10).until(EC.element_to_be_clickable((By.CLASS_NAME, "fxc-gcflink-link")))
        
        # Capture screenshot of the next dashboard
        driver.save_screenshot(f'next_dashboard_{index}.png')
        
    except Exception as e:
        print(f"Error occurred while processing alert {index}: {e}")
        # Scroll to the element
        actions = ActionChains(driver)
        actions.move_to_element(alert_link).perform()
        
        # Try clicking again
        alert_link.click()
        
        # Wait for the link to be clickable
        WebDriverWait(driver, 10).until(EC.element_to_be_clickable((By.CLASS_NAME, "fxc-gcflink-link")))
        
        # Capture screenshot of the next dashboard
        driver.save_screenshot(f'next_dashboard_{index}.png')
    
    # Go back to the previous page (the alert details)
    driver.back()
    
    # Wait for page to load
    time.sleep(5)

# Quit the driver
driver.quit()
