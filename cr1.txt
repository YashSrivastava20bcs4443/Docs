from selenium.webdriver.common.by import By
import time

# Find all alert elements by class name
alert_elements = driver.find_elements(By.CLASS_NAME, "fxc-gc-row")

# Iterate through each alert element
for index, alert_element in enumerate(alert_elements, 1):
    # Skip opening the link if its index is odd
    if index % 2 == 0:
        continue

    # Find the second link within the alert element
    link = alert_element.find_element(By.CLASS_NAME, "fxc-gcflink-link")
    
    # Click on the link to navigate to the affected resource page
    link.click()

    # Wait for the page to load
    time.sleep(5)

    # Capture screenshot of the affected resource page
    driver.save_screenshot(f'affected_resource_{index}.png')

    # Go back to the previous page (the alert details)
    driver.back()

    # Wait for page to load
    time.sleep(5)

# Quit the driver
driver.quit()
