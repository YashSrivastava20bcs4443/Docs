trigger:
- main  # Or specify the branch you want to trigger the pipeline

  vmImage: 'ubuntu-latest'  # Use an appropriate VM image based on your requirements

- task: UsePythonVersion@0
  inputs:
    versionSpec: '3.x'  # Specify the Python version you want to use

- script: |
    pip install -r requirements.txt  # Install Python dependencies from requirements.txt
  displayName: 'Install Python dependencies'

- script: |
    python your_script.py  # Replace 'your_script.py' with the name of your Python script file
  displayName: 'Run Python script'
