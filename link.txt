---
- name: Backup Fortinet firewall configuration
  hosts: fortigate
  gather_facts: no
  tasks:
    - name: Execute backup command
      command: >
        sshpass -p "{{ fortinet_password }}" ssh "{{ fortinet_username }}"@"{{ ansible_host }}"
        "execute backup config flash backup-{{ lookup('pipe', 'date +%Y-%m-%d') }}.conf"
      register: backup_result

    - name: Print backup result
      debug:
        msg: "{{ backup_result }}"
